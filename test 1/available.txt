  <script>
    // Function to show live time and date
    function updateDateTime() {
      const now = new Date();
      const options = { timeZone: "Asia/Kolkata", weekday: "long", year: "numeric", month: "long", day: "numeric", hour: "2-digit", minute: "2-digit", second: "2-digit" };
      const formatter = new Intl.DateTimeFormat("en-IN", options);
      document.getElementById("datetime").innerText = formatter.format(now);
    }
    setInterval(updateDateTime, 1000);

    // Edit availability status
    function editAvailability() {
      const availability = prompt("Enter new availability status (e.g., Available in Lab):");
      if (availability) {
        const statusElement = document.getElementById("availability-status");
        if (availability.toLowerCase().includes("available")) {
          statusElement.className = "availability-status available";
          statusElement.innerText = availability;
        } else {
          statusElement.className = "availability-status not-available";
          statusElement.innerText = availability;
        }
      }
    }
  </script>
